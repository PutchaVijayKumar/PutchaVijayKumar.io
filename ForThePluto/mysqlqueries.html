To select all data from expense type table

SELECT * FROM expense_type;


To select all data from expense_register table

SELECT * FROM expense_register;

To get the total sum of expense register

SELECT SUM(amount)
FROM expense_register;


To get the total sum of expense register based on expense type SELECT SUM(amount)
FROM expense_register
WHERE expense_type='17';


To inner join both tables and order based on date in descending order

SELECT expense_date, expense_type.expense_type, amount, description
FROM expense_register
INNER JOIN expense_type
ON expense_register.expense_type = expense_type.expense_id
ORDER BY expense_date DESC;


To inner join both tables and order by DATE in descending order and get data only for a particular expense type

SELECT expense_date, expense_type.expense_type, amount, description
FROM expense_register
INNER JOIN expense_type
ON expense_register.expense_type = expense_type.expense_id
WHERE expense_type.expense_type='Home Improvement'
ORDER BY expense_date DESC;


To inner join both tables and order by date in descending order only for a particular record type and where description starts with ___

SELECT expense_date, expense_type.expense_type, amount, description
FROM expense_register
INNER JOIN expense_type
ON expense_register.expense_type = expense_type.expense_id
WHERE expense_type.expense_type='EMI' AND expense_register.description LIKE 'P%'
ORDER BY expense_date DESC;


To get the sum of the amount column using inner join an filtering the data using expense type and expense description that starts with ___

SELECT SUM(amount)
FROM expense_register
INNER JOIN expense_type
ON expense_register.expense_type = expense_type.expense_id
WHERE expense_type.expense_type = 'EMI' AND expense_register.description LIKE 'P%';
